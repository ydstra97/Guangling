local function Cleanup()

    pcall(function() if G.flyEnabled and StopFly then StopFly() end end)
    pcall(function() if G.fbEnabled and RestoreLighting then RestoreLighting() end end)

    if psWorks then
        pcall(function() PS:UnregisterCollisionGroup("GuanglingPhantom") end)
    end
    
    pcall(function() charPartsCache = {} end)

        if stopWSLoop then stopWSLoop() end

            local char = lp and lp.Character
            local hum  = char and char:FindFirstChildOfClass("Humanoid")
            if hum and G.original.walkSpeed then hum.WalkSpeed = G.original.walkSpeed end
        end
    end)
    for _, c in ipairs(G.connections) do pcall(function() c:Disconnect() end) end
    G.connections = {}
    local old = CoreGui:FindFirstChild("GuanglingGUI")
    if old then old:Destroy() end
    _G.Guangling.Cleanup = nil
    _G.Guangling.version = nil
end

local function CaptureLighting()
    local data = { 
        Brightness = Lighting.Brightness, Ambient = Lighting.Ambient, OutdoorAmbient = Lighting.OutdoorAmbient, 
        FogEnd = Lighting.FogEnd, FogStart = Lighting.FogStart, ClockTime = Lighting.ClockTime, 
        ExposureCompensation = Lighting.ExposureCompensation, GlobalShadows = Lighting.GlobalShadows, effects = {} 
    }
    for _, v in ipairs(Lighting:GetChildren()) do 
        if v:IsA("PostEffect") then data.effects[v] = v.Enabled end 
    end
    return data
end

task.spawn(function()
    if not LocalPlayer.Character then LocalPlayer.CharacterAdded:Wait() end
    task.wait(1)
    if not G.original.lighting then G.original.lighting = CaptureLighting() end
end)

RestoreLighting = function()
    local o = G.original.lighting or CaptureLighting()
    G.original.lighting = o
    Lighting.Brightness = o.Brightness; Lighting.Ambient = o.Ambient; Lighting.OutdoorAmbient = o.OutdoorAmbient
    Lighting.FogEnd = o.FogEnd; Lighting.FogStart = o.FogStart; Lighting.ClockTime = o.ClockTime
    Lighting.ExposureCompensation = o.ExposureCompensation; Lighting.GlobalShadows = o.GlobalShadows
    for v, state in pairs(o.effects) do pcall(function() v.Enabled = state end) end
end

local wsLoopCon, wsEvtCon, wsCharCon
local currentWS = G.original.walkSpeed

local function applySpeed(speed)
    if not G.original.wsWasApplied then return end
    local hum = getHumanoid()
    if hum and hum.Parent and hum.Health > 0 then 
        if hum.WalkSpeed ~= speed then hum.WalkSpeed = speed end 
    end
end

local function bindWsHumanoid(hum, speed)
    if wsEvtCon then pcall(function() wsEvtCon:Disconnect() end) end
    wsEvtCon = hum:GetPropertyChangedSignal("WalkSpeed"):Connect(function() applySpeed(speed) end)
    conn(wsEvtCon)
end

stopWSLoop = function()
    G.original.wsWasApplied = false
    if wsLoopCon then pcall(function() wsLoopCon:Disconnect() end) wsLoopCon = nil end
    if wsEvtCon then pcall(function() wsEvtCon:Disconnect() end) wsEvtCon = nil end
    if wsCharCon then pcall(function() wsCharCon:Disconnect() end) wsCharCon = nil end
end

local function startWSLoop(speed)
    stopWSLoop()
    G.original.wsWasApplied = true
    G.original.lastWalkSpeed = speed
    local hum = getHumanoid()
    if hum then bindWsHumanoid(hum, speed) end
    wsCharCon = LocalPlayer.CharacterAdded:Connect(function(char)
        local newHum = char:WaitForChild("Humanoid")
        bindWsHumanoid(newHum, speed)
    end)
    conn(wsCharCon)
    wsLoopCon = RunService.Heartbeat:Connect(function()
        if not G.original.wsWasApplied then stopWSLoop() return end
        applySpeed(speed)
    end)
    conn(wsLoopCon)
end

local function StartFly()
    local rp, hum = getRootPart(), getHumanoid()
    if not rp or not hum or characterChanging then return end
    G.flyEnabled = true
    flyToggleBtn.BackgroundColor3 = Color3.fromRGB(200,60,60)
    flyToggleBtn.Text = "stop"
    
    cleanupFlyEngine()

    if flyMode == "Modern" then
        hum.PlatformStand = true 
        rp.Anchored = false -- V19.0.2 FIX: JANGAN ANCHORED (Hindari Anticheat Flag)
        hum.AutoRotate = false
        
        -- V19.0.2 FIX: Inject BodyForce penangkal gravitasi untuk Bypass Anticheat
        flyBV = Instance.new("BodyForce")
        flyBV.Name = "ModernAntiGravity"
        flyBV.Force = Vector3.new(0, workspace.Gravity * rp.AssemblyMass, 0)
        flyBV.Parent = rp
    else
        hum.PlatformStand = false
        rp.Anchored = false
        hum.AutoRotate = false
        flyBV = Instance.new("BodyVelocity")
        flyBV.Name = "MoverVel"
        flyBV.Velocity = Vector3.zero
        flyBV.MaxForce = Vector3.new(1e5, 1e5, 1e5)
        flyBV.Parent = rp
        
        flyBG = Instance.new("BodyGyro")
        flyBG.Name = "MoverGyro"
        flyBG.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
        flyBG.D = 50
        flyBG.Parent = rp
        
        task.spawn(function()
            while G.flyEnabled and flyMode == "Legacy" and flyBV and flyBG do
                task.wait(5)
                if flyBV then flyBV.Name = "MoverVel_" .. tostring(math.random(1000, 9999)) end
                if flyBG then flyBG.Name = "MoverGyro_" .. tostring(math.random(1000, 9999)) end
            end
        end)
    end

    flyCon = RunService.RenderStepped:Connect(function(dt)
        if not G.flyEnabled or characterChanging then return end
        local currentRP = getRootPart()
        local currentHum = getHumanoid()
        if not currentRP or not currentRP.Parent or not currentHum then StopFly(); return end
        
        local cam = workspace.CurrentCamera
        local spd = FLY_BASE * flySpeedLevel
        local dir = Vector3.zero
        
        if held.W then dir += cam.CFrame.LookVector end
        if held.S then dir -= cam.CFrame.LookVector end
        if held.D then dir += cam.CFrame.RightVector end
        if held.A then dir -= cam.CFrame.RightVector end
        if held.UP then dir += Vector3.new(0,1,0) end
        if held.DOWN then dir -= Vector3.new(0,1,0) end
        
        local targetVel = dir.Magnitude > 0 and dir.Unit * spd or Vector3.zero

        if flyMode == "Modern" then
            currentHum.PlatformStand = true
            currentRP.Anchored = false
            
            -- V19.0.2 FIX: Dynamic Velocity Matching (Server melihatmu bergerak legal)
            currentRP.AssemblyLinearVelocity = targetVel
            
            if targetVel.Magnitude > 0 then
                local offset = targetVel * dt
                local targetCF = CFrame.new(currentRP.Position + offset, currentRP.Position + offset + cam.CFrame.LookVector)
                currentRP.CFrame = currentRP.CFrame:Lerp(targetCF, 0.4)
            else
                local targetCF = CFrame.new(currentRP.Position, currentRP.Position + cam.CFrame.LookVector)
                currentRP.CFrame = currentRP.CFrame:Lerp(targetCF, 0.4)
            end
        else
            if flyBV and flyBV.Parent then 
                flyBV.Velocity = targetVel + Vector3.new(0, (not held.UP and not held.DOWN) and 0.5 or 0, 0) 
            end
            if flyBG and flyBG.Parent then flyBG.CFrame = cam.CFrame end
        end
    end)
    conn(flyCon)
end

local espRow = CreateRow()
CreateLabel(espRow, "ESP Highlight")

local ESP_COLOR    = Color3.fromRGB(255, 215, 0)
local MAX_DISTANCE = 500
local rayParams = RaycastParams.new()
rayParams.FilterType = Enum.RaycastFilterType.Exclude

local function removeESPForPlayer(player)
    local data = espCache[player]
    if not data then return end
    if data.playerConns then for _, c in ipairs(data.playerConns) do pcall(function() c:Disconnect() end) end end
    pcall(function() if data.highlight and data.highlight.Parent then data.highlight:Destroy() end end)
    pcall(function() if data.selBox and data.selBox.Parent then data.selBox:Destroy() end end)
    pcall(function() if data.label and data.label.Parent then data.label:Destroy() end end)
    espCache[player] = nil
end

clearESP = function() for player in pairs(espCache) do removeESPForPlayer(player) end end

local function isVisible(hrp)
    local rp = getRootPart()
    if not rp or not rp.Parent or not hrp or not hrp.Parent then return false end
    rayParams.FilterDescendantsInstances = {LocalPlayer.Character, hrp.Parent}
    return workspace:Raycast(rp.Position, hrp.Position - rp.Position, rayParams) == nil
end

local function getDepthMode(priority) return priority >= 3 and Enum.HighlightDepthMode.AlwaysOnTop or Enum.HighlightDepthMode.Occluded end

local function createESPForPlayer(player)
    if player == LocalPlayer or not player.Parent then return end
    if espCache[player] then removeESPForPlayer(player) end

    local playerConns = {}
    local data = { highlight = nil, selBox = nil, label = nil, txt = nil, hrp = nil, lastRaycast = 0, lastTextUpdate = 0, isVisible = true, playerConns = playerConns, isAttaching = true }
    espCache[player] = data

    local function attach(char)
        if not G.espEnabled then data.isAttaching = false return end
        if not char or char.Parent ~= workspace then data.isAttaching = false return end
        
        local hrp = char:WaitForChild("HumanoidRootPart", 3)
        if not hrp then data.isAttaching = false return end

        if data.highlight then data.highlight:Destroy() end
        if data.selBox then data.selBox:Destroy() end
        if data.label then data.label:Destroy() end

        local highlight = Instance.new("Highlight")
        highlight.FillColor = ESP_COLOR
        highlight.OutlineColor = ESP_COLOR
        highlight.FillTransparency = 0.8
        highlight.OutlineTransparency = 0
        highlight.Adornee = char
        highlight.Parent = char

        local selBox = Instance.new("SelectionBox")
        selBox.Color3 = ESP_COLOR
        selBox.LineThickness = 0.05
        selBox.SurfaceTransparency = 1
        selBox.Adornee = char
        selBox.Parent = CoreGui 

        local bill = Instance.new("BillboardGui")
        bill.Size = UDim2.new(0, 150, 0, 40)
        bill.StudsOffset = Vector3.new(0, 3, 0)
        bill.AlwaysOnTop = true
        bill.Adornee = hrp
        bill.Parent = CoreGui 

        local txt = Instance.new("TextLabel")
        txt.Size = UDim2.new(1, 0, 1, 0)
        txt.BackgroundTransparency = 1
        txt.Text = player.Name
        txt.TextColor3 = ESP_COLOR
        txt.TextStrokeTransparency = 0.2
        txt.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
        txt.Font = Enum.Font.GothamBold
        txt.TextSize = 11
        txt.RichText = true
        txt.Parent = bill

        data.highlight = highlight
        data.selBox = selBox
        data.label = bill
        data.txt = txt
        data.hrp = hrp
        data.isAttaching = false
    end

    if player.Character then task.defer(function() attach(player.Character) end) end
    table.insert(playerConns, player.CharacterAdded:Connect(attach))
end

local lastUpdate = 0
local espValidateTimer = 0

conn(RunService.Heartbeat:Connect(function(dt)
    if not G.espEnabled then return end
    local nowTime = tick()
    if nowTime - lastUpdate < 0.12 then return end
    lastUpdate = nowTime
    
    if nowTime - espValidateTimer > 2 then
        espValidateTimer = nowTime
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LocalPlayer and p.Character then
                local hrp = p.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local data = espCache[p]
                    if (not data or not data.highlight or not data.highlight.Parent or data.hrp ~= hrp) and (not data or not data.isAttaching) then
                        createESPForPlayer(p)
                    end
                end
            end
        end
    end

    local rp = getRootPart()
    if not rp or not rp.Parent then return end

    for player, data in pairs(espCache) do
        if not player.Parent or not player.Character then removeESPForPlayer(player); continue end
        if data.isAttaching then continue end
        
        local hrp = player.Character:FindFirstChild("HumanoidRootPart")
        if hrp and hrp.Parent and data.highlight then
            local dist = (hrp.Position - rp.Position).Magnitude
            if dist <= MAX_DISTANCE then
                
                -- V19.0.2 FIX: Frustum Culling & Proximity Throttling
                local cam = workspace.CurrentCamera
                local _, onScreen = cam:WorldToViewportPoint(hrp.Position)
                
                if onScreen and dist <= 200 then
                    if nowTime - data.lastRaycast > 0.25 then 
                        data.isVisible = isVisible(hrp)
                        data.lastRaycast = nowTime 
                    end
                else
                    -- Jangan habiskan CPU untuk mengecek dinding pada pemain di belakang kamera / terlalu jauh!
                    data.isVisible = false 
                end
                
                local head = player.Character:FindFirstChild("Head")
                local isHidden = (head and head.Transparency > 0.5) or false
                
                if isHidden then
                    data.highlight.Enabled = false
                    if data.selBox then data.selBox.Visible = true end
                else
                    data.highlight.DepthMode = getDepthMode((not data.isVisible and 4) or (dist < 50 and 2) or 1)
                    data.highlight.Enabled = true
                    if data.selBox then data.selBox.Visible = false end
                end

                if data.label and data.txt then
                    data.label.Enabled = true
                    if nowTime - data.lastTextUpdate > 0.5 then
                        data.lastTextUpdate = nowTime
                        local hum = player.Character:FindFirstChildOfClass("Humanoid")
                        local hp = hum and math.floor(hum.Health) or 0
                        local maxHp = hum and math.floor(hum.MaxHealth) or 100
                        local hpColor = (hp / maxHp > 0.5) and "<font color='rgb(50,255,50)'>" or "<font color='rgb(255,50,50)'>"
                        data.txt.Text = string.format("%s\n<font color='rgb(200,200,200)'>[%d m]</font> | %sHP: %d</font>", player.Name, math.floor(dist), hpColor, hp)
                    end
                end
            else
                data.highlight.Enabled = false
                if data.selBox then data.selBox.Visible = false end
                if data.label then data.label.Enabled = false end
            end
        end
    end
end))
